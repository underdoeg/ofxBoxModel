86 
87  //loop all blocks

89  std::string block = stringTrim(*itBlock);
90  if(block.size() != 0) {
91  std::vector<std::string> addressAndProperties = stringSplit(*itBlock, '{');
92  if(addressAndProperties.size() == 2) {

95  string address = stringTrim(addressAndProperties[0]);
96  string propertiesString = addressAndProperties[1];
97  std::vector<std::string> propertyList = stringSplit(propertiesString, ';');
98

101  std::vector<std::string> keyAndValue= stringSplit(*itProp, ':');
102

187  Color parseColor(std::string val) {
188  int r,g,b,a;
189  r = g = b = 0;
190  a = 255;
191  //convert hex color with #
192  if(stringContains(val, "#")) {
193  val = stringTrim(stringReplace(val, "#", ""));
194  if(val.size() == 2) {
195  std::string valNew;
196  valNew += val[0];
197  valNew+=val[1];
198  valNew+=val[0];
199  valNew+=val[1];
200  valNew+=val[0];
201  valNew+=val[1];
202  val = valNew;
203  } else if(val.size() == 3) { //check for color short codes
204  std::string valNew;
205  valNew += val[0];
206  valNew+=val[0];
207  valNew+=val[1];
208  valNew+=val[1];
209  valNew+=val[2];
210  valNew+=val[2];
211  val = valNew;
212  } else if(val.size() == 4) { //check for color short codes
213  std::string valNew;
214  valNew += val[0];
215  valNew+=val[0];
216  valNew+=val[1];
217  valNew+=val[1];
218  valNew+=val[2];
219  valNew+=val[2];
220  valNew+=val[3];
221  valNew+=val[3];
222  val = valNew;
223  }
224  long int number = (int)strtol(val.c_str(), NULL, 16);
225  if(val.size() == 6) {
226  r = ((number >> 16) & 0xFF);
227  g = ((number >> 8) & 0xFF);
228  b = ((number) & 0xFF);
229  a = 255;
230  } else if(val.size() == 8) {
231  r = ((number >> 32) & 0xFF);
232  g = ((number >> 16) & 0xFF);
233  b = ((number >> 8) & 0xFF);
234  a = ((number) & 0xFF);;
235  }
236  return Color(r, g, b, a);
237  } else if(stringContains(val, "rgb")) { //for rgb & rgba
238  std::vector<std::string> rgbAndNums = stringSplit(val, '(');
239  if(rgbAndNums.size()>1) {
240  std::vector<std::string> cols = stringSplit(stringReplace(rgbAndNums[1], ")", ""), ',');
241  if(cols.size()>0)
242  r = stringToInt(cols[0]);
243  if(cols.size()>1)
244  g = stringToInt(cols[1]);
245  if(cols.size()>2)
246  b = stringToInt(cols[2]);
247  if(cols.size()>3)
248  a = stringToInt(cols[3]);
249  } else {
250  debug::warning("css could not parse color "+val);
251  }
252  }
253  return Color(r, g, b, a);
254  }
